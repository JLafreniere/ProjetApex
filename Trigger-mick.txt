--Ajouter Champs Prénom Contact
CREATE OR REPLACE TRIGGER TI_Fournisseurs 
before insert
on Fournisseurs
for each row
BEGIN
:New.ID_Fournisseur := Seq_fournisseurs.nextval;
:New.Entreprise := InitCap(:New.Entreprise);
:New.Code_Postal := UPPER(:New.Code_Postal);
:New.Nom_Contact := InitCap(:New.Nom_Contact);

END TI_Fournisseurs;
/
